  export const daemonConfigSchema = z.object({
    version: z.literal(1),
    roots: z.array(z.string().min(1)).min(1),        // absolute Windows paths — the universe

    decision: z.enum(["allow","deny"]),
    createdAt: z.string(), origin: z.enum(["perm.decision","seed","cli"]) }).strict();
  export const allowlistFileSchema = z.object({ version: z.literal(1),
    rules: z.array(permissionRuleSchema) }).strict();
  export type AllowlistStore = {
    readonly rules: readonly PermissionRule[];
    match(q: { tool: PermissionRule["tool"]; scope: string }): "allow" | "deny" | "none";
    append(rule: PermissionRule): Promise<AllowlistStore>;   // immutable — returns NEW store
  };
  export async function loadAllowlist(filePath: string): Promise<AllowlistStore>;

  // src/permissions/broker.ts
  export type Verdict = { kind: "allow" }
    | { kind: "deny"; code: "outside_roots" | "permission_denied" | "permission_timeout";
        message: string };
  export type AskFn = (req: PermRequestPayload) =>
    Promise<{ allow: boolean; remember: boolean } | null>;   // null = nobody answered (timeout)
  export function createPermissionBroker(opts: { config: DaemonConfig; store: AllowlistStore;
    ask: AskFn; audit: AuditLog }): PermissionBroker;
  export type PermissionBroker = {
    decide(q: { tool: PermissionRule["tool"]; risk: Risk; scope: string;
                pathsToCheck: readonly string[] }): Promise<Verdict>;
  };
  // decide() order — LAW, in this order, no reordering:
  //   1. canonicalize every pathsToCheck entry; any failure -> deny (message, code outside_roots)
  //   2. isInsideRoots for each; any outside -> deny(outside_roots) WITHOUT asking — the hard

  //   3. store.match — explicit deny -> deny(permission_denied); allow -> allow

  // NEVER log file contents or the token; log byte counts.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| tool args (remote client) → local filesystem | every path string in a tool.request is hostile until canonicalized + roots-checked |
| allowlist file on disk → broker decisions | the store is state that grants power; corruption or tolerance = privilege |
| perm.request/decision loop → persisted grants | a UI click tonight is a standing grant tomorrow |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-65-05 | Elevation of privilege | paths.ts | mitigate | Canonicalize via `path.resolve` + `fs.realpath` (deepest-existing-ancestor variant for absent leaves), case-fold on win32, boundary check via `path.relative(root, target)` rejecting `..`-prefixed/absolute results — never `startsWith` on raw strings. NUL/ADS/UNC/device names rejected at canonicalize. Every hazard in the interfaces table is a named red-first test. |
| T-65-06 | Elevation of privilege | broker.ts ordering | mitigate | outside_roots is checked BEFORE the allowlist and is never promptable — no allowlist rule, however broad, can grant a path outside roots (rule scopes are themselves canonical paths inside roots, but the check does not rely on that). Order pinned by tests. |
| T-65-07 | Tampering | store.ts | mitigate | Atomic persistence: write `<file>.tmp` then `rename` (same-volume atomic on NTFS). Corrupt/unparseable file → fail CLOSED: back it up to `<file>.corrupt-<ts>`, start with EMPTY rules (zero remembered allows), log loudly. zod-validated on load; `.strict()` everywhere. |
| T-65-08 | Repudiation | audit.ts | mitigate | Every verdict (incl. denies and timeouts) and every execution is an append-only JSONL line with ts/tool/scope/verdict. No contents, no token — byte counts only. |
| T-65-09 | Denial of service | broker ask loop | mitigate | Unanswered ask resolves to DENY at `permTimeoutMs` (default 30s); pending asks are bounded per connection in 65-03. Silence is never consent. |
| T-65-10 | Information disclosure | config/state files | mitigate | `apps/daemon/.gitignore` ignores `daemon.config.json` (real local paths) and `.state/` (allowlist + audit). Only `daemon.config.example.json` is committed. No secrets live in config — the token is env-only. |
| T-65-SC | Tampering | npm installs | mitigate | No lane-side installs. All imports resolve against lockfile-pinned hoisted packages (zod 3.25.76, @types/node 20.19.43). `@types/ws` is declared for merge but unused in this plan. |
</threat_model>
