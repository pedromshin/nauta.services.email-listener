# Plan 03-02: SNS Notification Handler in FastAPI

## Goal
Add `POST /v1/emails/inbound-sns` — an unauthenticated route that handles the two SNS message
types sent by SES:
1. **SubscriptionConfirmation** — GETs the `SubscribeURL` to confirm the SNS subscription
2. **Notification** — parses the SES email metadata and logs it

No database writes in this phase. Success = email arrives, service logs `"email_received"` with
sender, recipient, and subject. The original `POST /v1/emails/inbound` endpoint is unchanged.

## Deployment order
Deploy this plan **before** 03-01's SNS subscription becomes active. Once this code is live on ECS:
- SNS will retry the PendingConfirmation POST
- The endpoint will GET the SubscribeURL → subscription becomes Confirmed
- Real email delivered to SES will trigger Notification POSTs → logged

## Pre-conditions
- Plan 03-01 applied: SES receipt rule active, SNS topic and subscription created
- Service is running on ECS Fargate behind the ALB (Phase 2 complete)
- `httpx` available (add if missing)

## SNS Notification Format

SNS sends `POST /v1/emails/inbound-sns` with `Content-Type: text/plain` and JSON body:

**SubscriptionConfirmation:**
```json
{
  "Type": "SubscriptionConfirmation",
  "Token": "...",
  "SubscribeURL": "https://sns.amazonaws.com/...",
  "TopicArn": "arn:aws:sns:us-east-1:..."
}
```

**Notification (SES inbound):**
```json
{
  "Type": "Notification",
  "MessageId": "sns-msg-id",
  "Message": "{\"mail\":{\"messageId\":\"...\",\"source\":\"...\",\"destination\":[\"...\"],\"commonHeaders\":{\"subject\":\"...\"}},\"content\":\"<base64-raw-email>\"}",
  "Timestamp": "2026-06-11T10:00:00.000Z"
}
```

`Message` is a JSON string. Parse it to extract `mail.messageId`, `mail.source`, `mail.destination`,
`mail.commonHeaders.subject`. Ignore `content` (raw MIME) in this phase.

```python
from __future__ import annotations
import logging
import httpx

logger = logging.getLogger(__name__)


async def confirm_subscription(subscribe_url: str) -> None:
    """GET the SubscribeURL to confirm an SNS HTTP subscription."""
    async with httpx.AsyncClient(timeout=10) as client:
        response = await client.get(subscribe_url)
        response.raise_for_status()
    logger.info("sns_subscription_confirmed url=%s", subscribe_url)
```

### Task 3 — SES notification parser

Create `apps/email-listener/app/infrastructure/sns/ses_parser.py`:

```python
from __future__ import annotations
import json
import logging
from typing import TypedDict

logger = logging.getLogger(__name__)


class EmailMeta(TypedDict):
    message_id: str
    sender: str
    recipients: list[str]
    subject: str


def parse_ses_notification(sns_message_str: str) -> EmailMeta:
    """Parse SES notification JSON from the SNS Message field."""
    data: dict = json.loads(sns_message_str)
    mail = data.get("mail", {})
    headers = mail.get("commonHeaders", {})
    return EmailMeta(
        message_id=mail.get("messageId", "unknown"),
        sender=mail.get("source", "unknown"),
        recipients=mail.get("destination", []),
        subject=headers.get("subject", "(no subject)"),
    )
```

### Task 4 — New unauthenticated SNS router

Create `apps/email-listener/app/presentation/routers/sns_inbound.py`:

```python
from __future__ import annotations
import json
import logging
from fastapi import APIRouter, Request, Response, status
from app.infrastructure.sns.confirmation import confirm_subscription
from app.infrastructure.sns.ses_parser import parse_ses_notification

logger = logging.getLogger(__name__)
router = APIRouter(prefix="/v1/emails", tags=["emails-sns"])


@router.post("/inbound-sns", status_code=status.HTTP_200_OK)
async def receive_inbound_sns(request: Request) -> Response:
    """Handle SNS notifications from SES. No auth — SNS cannot send X-API-Key."""
    raw = await request.body()
    try:
        payload: dict = json.loads(raw)
    except json.JSONDecodeError:
        logger.warning("sns_bad_json body=%r", raw[:200])
        return Response(status_code=status.HTTP_200_OK)  # return 200 to avoid SNS retry storm

    msg_type: str = payload.get("Type", "")

    if msg_type == "SubscriptionConfirmation":
        await confirm_subscription(payload["SubscribeURL"])
        return Response(status_code=status.HTTP_200_OK)

    if msg_type == "Notification":
        try:
            meta = parse_ses_notification(payload["Message"])
        except Exception:
            logger.exception("sns_parse_error payload=%r", payload)
            return Response(status_code=status.HTTP_200_OK)

        logger.info(
            "email_received message_id=%s sender=%s recipients=%s subject=%r",
            meta["message_id"],
            meta["sender"],
            meta["recipients"],
            meta["subject"],
        )
        return Response(status_code=status.HTTP_200_OK)

    logger.warning("sns_unknown_type type=%s", msg_type)
    return Response(status_code=status.HTTP_200_OK)
```

> Returns 200 for all cases (including errors) to prevent SNS from retrying indefinitely on
> malformed messages. Malformed inputs are logged as warnings, not errors.

### Task 6 — Tests

Create `apps/email-listener/tests/test_inbound_sns.py`:

```python
from __future__ import annotations
import json
import pytest
from httpx import AsyncClient, ASGITransport
from unittest.mock import AsyncMock, patch
from app.main import app


@pytest.mark.anyio
async def test_subscription_confirmation_calls_subscribe_url() -> None:
    payload = {
        "Type": "SubscriptionConfirmation",
        "Token": "token123",
        "SubscribeURL": "https://sns.amazonaws.com/confirm?token=token123",
        "TopicArn": "arn:aws:sns:us-east-1:123:topic",
    }
    with patch(
        "app.presentation.routers.sns_inbound.confirm_subscription",
        new_callable=AsyncMock,
    ) as mock_confirm:
        async with AsyncClient(transport=ASGITransport(app=app), base_url="http://test") as client:
            resp = await client.post(
                "/v1/emails/inbound-sns",
                content=json.dumps(payload),
                headers={"Content-Type": "text/plain"},
            )
        assert resp.status_code == 200
        mock_confirm.assert_awaited_once_with(payload["SubscribeURL"])


@pytest.mark.anyio
async def test_notification_logs_email_meta(caplog: pytest.LogCaptureFixture) -> None:
    ses_message = json.dumps({
        "mail": {
            "messageId": "msg-001",
            "source": "sender@example.com",
            "destination": ["agent@magnitudetech.com.br"],
            "commonHeaders": {"subject": "Hello from test"},
        }
    })
    payload = {
        "Type": "Notification",
        "Message": ses_message,
        "MessageId": "sns-001",
        "TopicArn": "arn:aws:sns:us-east-1:123:topic",
        "Timestamp": "2026-06-11T10:00:00.000Z",
    }
    async with AsyncClient(transport=ASGITransport(app=app), base_url="http://test") as client:
        resp = await client.post(
            "/v1/emails/inbound-sns",
            content=json.dumps(payload),
            headers={"Content-Type": "text/plain"},
        )
    assert resp.status_code == 200
    assert "email_received" in caplog.text
    assert "msg-001" in caplog.text


@pytest.mark.anyio
async def test_bad_json_returns_200() -> None:
    async with AsyncClient(transport=ASGITransport(app=app), base_url="http://test") as client:
        resp = await client.post(
            "/v1/emails/inbound-sns",
            content=b"not json",
            headers={"Content-Type": "text/plain"},
        )
    assert resp.status_code == 200
```
